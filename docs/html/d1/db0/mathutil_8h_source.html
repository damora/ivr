140 static inline void getfrustum(float fovy, float aspect, float *vv)
141 {
142  // use projection matrix to compute viewvolume extents
143  fovy = RADIANS * fovy;
144  float l,r,t,b;
145  t = tan(fovy/2.0f) * 5.0f; b = -t;
146  l = aspect * b; r = aspect * t;
147 
148  vv[0] = l, vv[1] = r; vv[2] = b; vv[3] = t;
149 }

158 static inline vector3f ndctowindow(vector3f v, WINDOW viewport)
159 {
160  float x, y, w, h;
161  float far, near;
162  vector3f win;
163 
164  // window xform
165  x = viewport.x;
166  y = viewport.y;
167  w = viewport.w;
168  h = viewport.h;
169  far = w; // could add a setdepthrange function and set near, far there, e.g. OGL
170  near = 0.0f;
171 
172  win.x = w/2.0f * v.x + (x + w/2.0f);
173  win.y = h/2.0f * v.y + (y + h/2.0f);
174  win.z = (far-near)/2.0f * v.z + (far+near)/2.0f;
175 
176  win.x = (win.x < x) ? x : win.x;
177  win.x = (win.x > x+w) ? x+w : win.x;
178  win.y = (win.y < y) ? y : win.y;
179  win.y = (win.y > y+h) ? y+h : win.y;
180 // win.z = (win.z < 0) ? 0 : win.z;
181 // win.z = (win.z > w) ? w : win.z;
182 
183  return win;
184 }
