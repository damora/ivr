90 static inline float getsamplevalue(float x, float y, float z)
91 {
92  float x0, y0, z0, xd, yd, zd, xp, yp, zp;
93 
94  // map x, y, z to volumetric data indices
95  vector3f xyz = {x, y, z};
96 
97  // volume dimensions
98  int vw = volume.localdims.w;
99  int vh = volume.localdims.h;
100  int vd = volume.localdims.d;
101 
102 #ifdef BGQ
103  float fvw = volume.fvoldims.x;
104  float fvh = volume.fvoldims.y;
105  float fvd = volume.fvoldims.z;
106 
107  float rfvw = volume.rfvoldims.x;
108  float rfvh = volume.rfvoldims.y;
109  float rfvd = volume.rfvoldims.z;
110 
111  float fj = __frizs (x * fvw/aabb.w - (aabb.min.x*fvw)/aabb.w);
112  float fi = __frizs (y * fvh/aabb.h - (aabb.min.y*fvh)/aabb.h);
113  float fk = __frizs (z * fvd/aabb.d - (aabb.min.z*fvd)/aabb.d);
114 
115  fj = __fsels((fj - (fvw-1.0f)), fvw-2.0f, fj);
116  fi = __fsels((fi - (fvh-1.0f)), fvh-2.0f, fi);
117  fk = __fsels((fk - (fvd-1.0f)), fvd-2.0f, fk);
118 
119  int i = (int) fi;
120  int j = (int) fj;
121  int k = (int) fk;
122 
123  PRINTDEBUG2("IJK: %d,%d,%d\n", i,j,k);
124 
125 #else
126  // number of voxels in each direction and should be set when volume is input
127  // mapping should be (volr-voll)/(box.r-box.l) + (aabb.x*vw/bboxw
128 
129  float ratio = (xyz.x-aabb.min.x)/aabb.w;
130  int j = vw * ratio;
131  ratio = (xyz.y-aabb.min.y)/aabb.h;
132  int i = vh * ratio;
133  ratio = (xyz.z-aabb.min.z)/aabb.d;
134  int k = vd * ratio;
135 
136 /*
137  int j = (int) (xyz.x * vw/aabb.w - (aabb.min.x*vw)/aabb.w);
138  int i = (int) (xyz.y * vh/aabb.h - (aabb.min.y*vh)/aabb.h);
139  int k = (int) (xyz.z * vd/aabb.d - (aabb.min.z*vd)/aabb.d);
140 */
141  /*
142  i = (i >= vh-1) ? vh-2 : i;
143  j = (j >= vw-1) ? vw-2 : j;
144  k = (k >= vd-1) ? vd-2 : k;
145 */
146  i = (i >= vh-1) ? vh-1 : i;
147  j = (j >= vw-1) ? vw-1 : j;
148  k = (k >= vd-1) ? vd-1 : k;
149 
150 #endif
151 
152  // get 8 neighboring voxels
153  float v000, v100, v010, v001, v101, v011, v110, v111;
154 
155  v000 = *(volume.values + maptoffset3d(i, j, k, vw, vh, vd));
156  v100 = *(volume.values + maptoffset3d(i+1, j, k, vw, vh, vd));
157  v010 = *(volume.values + maptoffset3d(i, j+1, k, vw, vh, vd));
158  v001 = *(volume.values + maptoffset3d(i, j, k+1, vw, vh, vd));
159  v101 = *(volume.values + maptoffset3d(i+1, j, k+1, vw, vh, vd));
160  v011 = *(volume.values + maptoffset3d(i, j+1, k+1, vw, vh, vd));
161  v110 = *(volume.values + maptoffset3d(i+1, j+1, k, vw, vh, vd));
162  v111 = *(volume.values + maptoffset3d(i+1, j+1, k+1, vw, vh, vd));
163 
164  // tri-linear interpolation; I should pre-compute x0, y0, z0 for each cell probably
165  // map to 0..aabb->max
166  xp = x + fabs(aabb.min.x);
167  yp = y + fabs(aabb.min.y);
168  zp = z + fabs(aabb.min.z);
169 #ifdef BGQ
170  x0 = (fj * aabb.w) * rfvw;
171  y0 = (fi * aabb.h) * rfvh;
172  z0 = (fk * aabb.d) * rfvd;
173 
174  xd = (__fabss(xp - x0) * aabb.w) * rfvw;
175  yd = (__fabss(yp - y0) * aabb.h) * rfvh;
176  zd = (__fabss(zp - z0) * aabb.d) * rfvd;
177 #else
178  x0 = (j * aabb.w)/vw;
179  y0 = (i * aabb.h)/vh;
180  z0 = (k * aabb.d)/vd;
181 
182  xd = (fabsf(xp - x0) * aabb.w)/vw;
183  yd = (fabsf(yp - y0) * aabb.h)/vh;
184  zd = (fabsf(zp - z0) * aabb.d)/vd;
185 #endif
186  float c00 = v000 * (1.0f - xd) + v100 * xd;
187  float c10 = v010 * (1.0f - xd) + v110 * xd;
188  float c01 = v001 * (1.0f - xd) + v101 * xd;
189  float c11 = v011 * (1.0f - xd) + v111 * xd;
190  float c0 = c00 * (1.0f - yd) + c10 * yd;
191  float c1 = c01 * (1.0f - yd) + c11 * yd;
192  float c = c0 * (1.0f - zd) + c1 * zd;
193 
194  return c;
195 }

197 int getsamplesonray(ray3f ray, int maxsamples, float *samples)
198 {
199 
200  float deltat = DELTASAMPLE;
201  float x, y, z;
202  float tin, tout;
203  boolean_t hit = FALSE;
204  if ((hit = getrayintersect(ray, &tin, &tout)) != TRUE) {
205  return -1;
206  }
207 
208  cnt++;
209  int i=0;
210  float t;
211  int iend = 1.0f + (tout - tin)/deltat + 0.5f;
212  iend = (iend < maxsamples) ? iend : maxsamples;
213 #pragma omp parallel for private(x, y, z, t) lastprivate(i)
214  for (i=0; i<iend; i++) {
215  t = tin + (i * deltat);
216 
217  x = ray.dir.x * t + ray.orig.x;
218  y = ray.dir.y * t + ray.orig.y;
219  z = ray.dir.z * t + ray.orig.z;
220 
221 
222  samples[i] = getsamplevalue(x, y, z);
223 #ifdef REVISIT // probably need to save these intersections for later isosurface painting...yuk
224  samples[i].intersect.x = x;
225  samples[i].intersect.y = y;
226  samples[i].intersect.z = z;
227 #endif
228  }
229  return i;
230 }

233 boolean_t getrayintersect(ray3f ray, float *tin, float* tout)
234 {
235  int i;
236  float numer, denom;
237  float thit;
238  vector3f * n;
239  boolean_t hit = FALSE;
240 
241  n = aabb.facenormals;
242 
243  *tin = -MAXFLOAT;
244  *tout = MAXFLOAT;
245  i = 0;
246  while ((i < MAXPLANE) && (*tin < *tout)) {
247  // compute numer, denom for ith plane
248  numer = pzero[i] - dot3f(n[i], ray.orig);
249  denom = dot3f(n[i], ray.dir);
250  if (denom == 0.0f) {
251  if (numer < 0.0f) *tin = *tout + 1.0f;
252  }
253  else {
254  thit = numer/denom;
255  if (denom > 0.0f ) {
256  *tout = fmin(*tout, thit);
257  }
258  else {
259  *tin = fmax(*tin, thit);
260  }
261  }
262  i++;
263  }
264  if (*tin < *tout) {
265  if (*tin > 0.0f) {
266  hit = TRUE;
267 // thit = *tin;
268  }
269  else if (*tout > 0.0f) {
270  hit = TRUE;
271 // thit = *tout;
272  }
273  }
274  return hit;
275 }

290 int getfirstvoxel(ray3f ray, vector3i *samples)
291 {
292  float times[6];
293  int hits[6];
294  int faces[6];
295  double t;
296  if (ray.dir.x==0) { times[0] = times[1] = 0.0; }
297  else {
298  t = aabb.min.x/ray.dir.x;
299  times[0] = t; faces[0] = X_FACE;
300  hits[0] = hitface(t*ray.dir.y , t*ray.dir.z , aabb.min.y , aabb.max.y , aabb.min.z , aabb.max.z);
301  t = aabb.max.x/ray.dir.x;
302  times[1] = t; faces[1] = X_FACE + MAX_FACE;
303  hits[1] = hitface(t*ray.dir.y , t*ray.dir.z , aabb.min.y , aabb.max.y , aabb.min.z , aabb.max.z);
304  }
305  if (ray.dir.y==0) { times[2] = times[3] = 0.0; }
306  else {
307  t = aabb.min.y/ray.dir.y;
308  times[2] = t; faces[2] = Y_FACE;
309  hits[2] = hitface(t*ray.dir.x , t*ray.dir.z , aabb.min.x , aabb.max.x , aabb.min.z , aabb.max.z);
310  t = aabb.max.y/ray.dir.y;
311  times[3] = t; faces[3] = Y_FACE + MAX_FACE;
312  hits[3] = hitface(t*ray.dir.x , t*ray.dir.z , aabb.min.x , aabb.max.x , aabb.min.z , aabb.max.z);
313  }
314  if (ray.dir.z==0) { times[4] = times[5] = 0.0; }
315  else {
316  t = aabb.min.z/ray.dir.z;
317  times[4] = t; faces[4] = Z_FACE;
318  hits[4] = hitface(t*ray.dir.x , t*ray.dir.y , aabb.min.x , aabb.max.x , aabb.min.y , aabb.max.y);
319  t = aabb.max.z/ray.dir.z;
320  times[5] = t; faces[5] = Z_FACE + MAX_FACE;
321  hits[5] = hitface(t*ray.dir.x , t*ray.dir.y , aabb.min.x , aabb.max.x , aabb.min.y , aabb.max.y);
322  }
323  int first = 6;
324  t = 0.0;
325  for (int i=0 ; i<6 ; i++) {
326  if (times[i] > 0.0 && (times[i]<t || t==0.0)) {
327  first = i;
328  t = times[i];
329  }
330  }
331  vector3f v;
332  v.x = ray.dir.x*times[first] + ray.orig.x;
333  v.y = ray.dir.y*times[first] + ray.orig.y;
334  v.z = ray.dir.z*times[first] + ray.orig.z;
335 
336  samples->x = (int) (v.x + 0.5f);
337  samples->y = (int) (v.y + 0.5f);
338  samples->z = (int) (v.z + 0.5f);
339  if (first>5) return -1; // Found nothing
340  else return 0; // Probably want hits[first] and faces[first] also....
341 }

344 void getvoxelsonray(ray3f ray, int maxdepth, float *samples)
345 {
346  // Implementation is based on:
347  // "A Fast Voxel Traversal Algorithm for Ray Tracing"
348  // John Amanatides, Andrew Woo
349  // http://www.cse.yorku.ca/~amana/research/grid.pdf
350  // http://www.devmaster.net/articles/raytracing_series/A%20faster%20voxel%20traversal%20algorithm%20for%20ray%20tracing.pdf
351 
352  // NOTES:
353  // * This code assumes that the ray's origition and direction are in 'cell coordinates', which means
354  // that one unit equals one cell in all directions.
355  // * When the ray doesn't start within the voxel grid, calculate the first origition at which the
356  // ray could enter the grid. If it never enters the grid, there is nothing more to do here.
357  // * Also, it is important to test when the ray exits the voxel grid when the grid isn't infinite.
358  // * The vector3f structure is a simple structure having three integer fields (X, Y and Z).
359 
360  // The cell in which the ray starts.
361 
362  int rc=0;
363  vector3i first;
364  // Rounds the origition's x, y and z down to the nearest integer values.
365  if ((rc=getfirstvoxel(ray, &first)) == -1) {
366  //fprintf(stderr, "ray misses volume\n");
367  return;
368  };
369  int x = first.x;
370  int y = first.y;
371  int z = first.z;
372 
373 
374  // Determine which way we go.
375  vector3i step;
376  step.x = signof(ray.dir.x);
377  step.y = signof(ray.dir.y);
378  step.z = signof(ray.dir.z);
379 
380  // Calculate cell boundaries. When the step (i.e. direction sign) is origitive,
381  // the next boundary is AFTER our current origition, meaning that we have to add 1.
382  // Otherwise, it is BEFORE our current origition, in which case we add nothing.
383  vector3f cellboundary;
384  cellboundary.x = x + (step.x > 0 ? 1 : 0);
385  cellboundary.y = y + (step.y > 0 ? 1 : 0);
386  cellboundary.z = z + (step.z > 0 ? 1 : 0);
387 
388 
389  // NOTE: For the following calculations, the result will be HUGE_VAL (+infinity)
390  // when ray.dir.x, y or z equals zero, which is OK. However, when the left-hand
391  // value of the division also equals zero, the result is Single.NaN, which is not OK.
392 
393  // Determine how far we can travel along the ray before we hit a voxel boundary.
394  vector3f tmax;
395  tmax.x = (cellboundary.x - ray.orig.x) / ray.dir.x; // Boundary is a plane on the YZ axis.
396  tmax.y = (cellboundary.y - ray.orig.y) / ray.dir.y; // Boundary is a plane on the XZ axis.
397  tmax.z = (cellboundary.z - ray.orig.z) / ray.dir.z; // Boundary is a plane on the XY axis.
398 
399  if (isnan(tmax.x)) tmax.x = HUGE_VAL;
400  if (isnan(tmax.y)) tmax.y = HUGE_VAL;
401  if (isnan(tmax.z)) tmax.z = HUGE_VAL;
402 
403  // Determine how far we must travel along the ray before we have crossed a gridcell.
404  vector3f deltat ;
405  deltat.x = step.x / ray.dir.x; // Crossing the width of a cell.
406  deltat.y = step.y / ray.dir.y; // Crossing the height of a cell.
407  deltat.z = step.z / ray.dir.z; // Crossing the depth of a cell.
408 
409  if (isnan(deltat.x)) deltat.x = HUGE_VAL;
410  if (isnan(deltat.y)) deltat.y = HUGE_VAL;
411  if (isnan(deltat.z)) deltat.z = HUGE_VAL;
412 
413  // For each step, determine which distance to the next voxel boundary is lowest (i.e.
414  // which voxel boundary is nearest) and walk that way.
415  for (int i = 0; i < maxdepth; i++)
416  {
417  // compute index and save it.
418  samples[i] = maptoffset3d(x,y,z, volume.localdims.w, volume.localdims.h, volume.localdims.d);
419 
420  // Do the next step.
421  if (tmax.x < tmax.y && tmax.x < tmax.z)
422  {
423  // tmax.x is the lowest, an YZ cell boundary plane is nearest.
424  x += step.x;
425  tmax.x += deltat.x;
426  }
427  else if (tmax.y < tmax.z)
428  {
429  // tmax.y is the lowest, an XZ cell boundary plane is nearest.
430  y += step.y;
431  tmax.y += deltat.y;
432  }
433  else
434  {
435  // tmax.z is the lowest, an XY cell boundary plane is nearest.
436  z += step.z;
437  tmax.z += deltat.z;
438  }
439  }
440 }
