264 int slicingcalc(int totsize, int minslicesize, int nslices, int log2align, int overlap, int rank, int *slicesize, int *sliceorigin)
265 {
266  // Given the total number of points in one dimension, the minimum number
267  // of points required for an MPI transfer, the number of bricks
268  // desired, the alignment required (0=no alignment, 1=multiple of 2
269  // elements, 2=multiple of 4, 3=multiple of 8, etc.), and the rank,
270  // calculate the size and origin of the rank'th bricks. Exit if alignment
271  // requested is not valid or if problem size is not a multiple of the
272  // requested alignment.
273  // N.B. "rank" runs from 0 to nslices-1.
274 
275  int mask, minsize, ns, bottom, top;
276 
277  if (nslices == 1) {
278  *sliceorigin = 0;
279  *slicesize = totsize;
280  return(0);
281  }
282 
283  // Alignment required must be none, 2, 4, 8, 16, or 32 elements; the
284  // code will work for more, but a larger request is likely to be an
285  // error, so this test is provided to make sure the programmer realizes
286  // what he/she has requested.
287  if (log2align<0 || log2align>5) {
288  printf("slicingCalc error: log2align = %d, must be in range 0 to 5\n",log2align);
289  return(-1);
290  }
291 
292  // For now, require each dimension to be an exact multiple of the required
293  // alignment. This could be relaxed by padding the input.
294  mask = (1<<log2align)-1;
295  if ((totsize&mask)!=0) {
296  printf("slicingCalc error: totsize = %d, must be a multiple of required alignment %d\n",totsize,1<<log2align);
297  return(-1);
298  }
299  // Minimum bricks size is stencil size rounded up to alignment boundary
300 
301  minsize = ((minslicesize+mask)>>log2align)<<log2align;
302  if ((minsize*nslices)>=totsize) {
303  // Must use bricks of the minimum size; use as many as needed, and
304  // do not use the remaining bricks. Note that the last bricks may be
305  // slightly larger.
306  ns=totsize/minsize; // number of bricks which can be made; note rounding down
307  if (rank<ns) {
308  *sliceorigin = rank*minsize;
309  if (rank<(ns-1)) {
310  *slicesize = minsize;
311  } else {
312  *slicesize=totsize-(rank*minsize);
313  }
314  } else {
315  *sliceorigin = -1;
316  *slicesize = -1;
317  }
318  } else {
319  // We have enough points to give some to all bricks; take the ideal
320  // boundaries and round them to the nearest aligned boundaries.
321 
322  bottom = rank*totsize/nslices; // ideal boundaries
323  top = (rank+1)*totsize/nslices;
324  bottom += (1<<log2align)>>1; // add half of alignment
325  top += (1<<log2align)>>1;
326  top = (top>>log2align)<<log2align; // round down to aligned boundary
327  bottom = (bottom>>log2align)<<log2align;
328  bottom = (rank != 0) ? bottom-overlap : bottom;
329  *sliceorigin = bottom;
330  *slicesize = top-bottom;
331  }
332  return 0;
333 }
